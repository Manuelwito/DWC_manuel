function inicializar(){
    const myInp = document.getElementById('username');
    const mensaje = document.getElementById('mensaje');
    myInp.addEventListener("blur", required);
   


    function required() { 
   
        if(myInp.value == ""){
            mensaje.innerHTML="rellenalo";
            myInp.focus();
        }
        
        else{
            alert("flama");
            mensaje.innerHTML="";
        }
    }
    
}
